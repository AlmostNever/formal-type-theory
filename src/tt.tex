    \infer
    { }
    {\isctx{{\ctxempty}}}
}

\newcommand{\rlCtxExtend}{\referTo{CtxExtend}{rul:CtxExtend}}

    \infer
    { \mathtt{precond:}\, \mathtt{isctx}\, {\Gamma} \\
\istype{{\Gamma}}{A}}
    {\isctx{\ctxextend{{\Gamma}}{A}}}
}

                           \end{equation}
