%%% Type theory

% Generic entities
\newcommand{\G}{\Gamma} % a context
\newcommand{\D}{\Delta} % another context
\newcommand{\A}{A} % a type
\newcommand{\B}{B} % another type
\newcommand{\C}{C} % yet another type
\newcommand{\x}{x} % a term variable
\newcommand{\y}{y} % another term variable
\newcommand{\uu}{u} % a term
\newcommand{\vv}{v} % another term
\newcommand{\ww}{w} % yet another term
\newcommand{\sbs}{\sigma} % a substitution

\newcommand{\rl}[1]{\text{\textsc{#1}}}

%% Syntax
\newcommand{\bnf}{\ \mathrel{{:}{:}{=}}\ }
\newcommand{\bnfor}{\ \mid\ \ }

%% Contexts
\newcommand{\ctxempty}{\bullet} % empty context
\newcommand{\ctxextend}[3]{#1,\, #2\, {:}\, #3} % extended context

\newcommand{\ctxdom}[1]{\mathsf{dom}(#1)} % the domain of a context

%% Substitution
\newcommand{\sbunit}[1]{\mathsf{1}_{#1}} % empty substitution
\newcommand{\sbextend}[3]{#1,#2{\mapsto}#3} % extended substitution
\newcommand{\subst}[2]{#1[#2]} % substitution operation

% Type formers
\newcommand{\Prod}[2]{\mathop{\textstyle\prod_{(#1 {:} #2)}}} % dependent product
\newcommand{\Id}[3]{\mathsf{Id}_{#1}(#2,#3)} % Identity type

% Terms
\newcommand{\lam}[3]{\lambda #1 {:} #2.{#3}\,.\,} % $\lambda$-abstraction
\newcommand{\app}[5]{#1\mathbin{@^{#2{:}#3.#4}} #5} % application
\newcommand{\refl}[1]{{\mathsf{refl}_{#1}}\ }    % reflexivity

% Judgments
\newcommand{\isctx}[1]{#1\ \mathsf{ctx}} % well formed context
\newcommand{\istype}[2]{#1 \vdash #2 \ \mathsf{type}} % well formed type
\newcommand{\isterm}[3]{#1 \vdash #2 : #3} % well formed term
\newcommand{\issubst}[3]{#1 : #2 \to #3} % well formed substitution

\newcommand{\eqctx}[2]{#1 \equiv #2 : \mathsf{ctx}}
\newcommand{\eqtype}[3]{#1 \vdash #2 \equiv #3 : \mathsf{type}} % equal types
\newcommand{\eqterm}[4]{#1 \vdash #2 \equiv #3 : #4} % equal terms

%%%% Translation

\newcommand{\defeq}{\mathrel{{:}{\equiv}}}

\newcommand{\Iso}[2]{#1 \cong #2}

\newcommand{\JJ}{\mathcal{J}} % any judgment

\newcommand{\DD}{\mathcal{D}} % a derivation
\newcommand{\EE}{\mathcal{E}} % another derivation
\newcommand{\derives}[2]{#1 \mathbin{{:}{:}} #2} % a derivation with its conclusion

\newcommand{\tran}[1]{\langle\!\langle #1 \rangle\!\rangle} % direct translation
\newcommand{\conn}[2]{\tran{#1} \rightleftharpoons \tran{#2}} % path between translations

\newcommand{\Case}[1]{\subsubsection{Case $\rl{#1}$}}
